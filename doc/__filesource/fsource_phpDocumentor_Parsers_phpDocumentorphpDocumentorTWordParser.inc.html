  226.

  232.             $gpos  = 0;
  233.             $found = false;
  234.             if ($this->tokenEquals($word, $this->_global_search[$gpos++])) {
  235.                 $found = true;
  236.                 for (;$gpos<count($this->_global_search);$gpos++, $pos++) {
  237.                     if (isset($this->_all[$pos]) &&
  238.                         !$this->tokenEquals($this->_global_search[$gpos], 
  239.                             $this->_all[$pos])
  240.                     ) {
  241.                         $found = false;
  242.                     }
  243.                 }

  245.             if ($found) {

  247.                 $this->pos += count($this->_global_search) - 1;

  256.             if (in_array($word[0], $this->_nl_check)) {
  257.                 $this->linenum += substr_count($word[1], "\n");
  258.             }
  259.             if ($word[0] == T_WHITESPACE && !$this->returnWhiteSpace) {
  260.                 return $this->getWord();
  261.             }
  262.             // seeing if we can get line numbers out of the beast
  263.         }
  264.         if (is_array($word) && $word[0] == T_COMMENT) {
  265.             $this->_docblock_linenum = $oldlinenum;
  266.         }
