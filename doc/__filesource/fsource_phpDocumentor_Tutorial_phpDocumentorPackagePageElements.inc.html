  444.     {
  445.         if (is_string($el)) {
  446.             return parent::add($el);
  447.         }
  448.         if (phpDocumentor_get_class($el) == 'parserxmldocbooktag') {
  449.             if ($el->name == 'title') {
  450.                 $this->setTitle($el);
  451.             } else {
  452.                 return parent::add($el);
  453.             }
  454.         } else {
  455.             return parent::add($el);
  456.         }
  457.     }
