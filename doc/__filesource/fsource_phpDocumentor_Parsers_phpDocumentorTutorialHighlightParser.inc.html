  358.     function checkEventPush($word, $pevent)
  359.     {
  360.         $e = false;
  361.         if (isset($this->pushEvent[$pevent])) {

  383.             return false;

  386.             return $this->_event_stack->popEvent();
  387.         } else {
  388.             return false;
