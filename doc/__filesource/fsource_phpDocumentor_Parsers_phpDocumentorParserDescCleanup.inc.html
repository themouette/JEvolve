  694.                    || $is_valid

  768.     function getStartList($word)

  786.         } else
  787.         {
  788.             $this->p_vars['last_list'] = $this->p_vars['start_list'];

  807.     /**#@+

  812.     /**
  813.      * Handles <<ol>>,<<li>>,<<ul>>
  814.      *
  815.      * This allows parsing of lists nested to any level.  Using
  816.      * the lists and list_item temporary variables and using
  817.      * list_count to control nesting, the method creates a {@link parserList}

  819.      * standard {@link parserStringWithInlineTags} for all the text, adding
  820.      * in nested lists as if they were inline tags (the conversion interface
  821.      * is the same for both object types)
  822.      */
  823.     function handleList($word, $pevent)
